```php
use App\Models\EconomicIndicator;
use App\Models\LearningUpdate;
use App\Models\ReferenceDataEntry;
use App\Models\User;
use App\Services\Learning\ApprovalFlow;
use App\Services\ReferenceData\ReferenceDataSubmission;
use App\Support\RequestContext;
use Illuminate\Support\Carbon;

app(RequestContext::class)->apply('system', ['reason' => 'test_live_reference_data_parity']);

$actor = User::query()
    ->where('email', 'itadmin@futureshiftadvisory.nz')
    ->first()
    ?? User::query()->role(User::TYPE_SUPER_ADMIN)->firstOrFail();

$items = [
    [
        'indicator' => 'cpi_annual',
        'label' => 'CPI annual',
        'value' => 3.1,
        'unit' => 'percent',
        'period_date' => '2026-06-01',
        'as_at' => '2026-06-11',
        'source' => 'manual_admin',
    ],
    [
        'indicator' => 'ocr',
        'label' => 'OCR reference rate',
        'value' => 2.25,
        'unit' => 'percent',
        'period_date' => '2026-06-11',
        'as_at' => '2026-06-11',
        'source' => 'manual_admin',
    ],
];

foreach ($items as $item) {
    $existing = ReferenceDataEntry::query()
        ->where('dataset', ReferenceDataEntry::DATASET_ECONOMIC_INDICATOR)
        ->latest()
        ->get()
        ->first(fn ($entry) => (string) data_get($entry->payload, 'indicator') === $item['indicator']
            && (string) data_get($entry->payload, 'period_date') === $item['period_date']
            && (float) data_get($entry->payload, 'value') === (float) $item['value']);

    $entry = $existing instanceof ReferenceDataEntry
        ? $existing
        : app(ReferenceDataSubmission::class)->submit(
            dataset: ReferenceDataEntry::DATASET_ECONOMIC_INDICATOR,
            payload: [
                'indicator' => $item['indicator'],
                'label' => $item['label'],
                'value' => $item['value'],
                'unit' => $item['unit'],
                'period_date' => $item['period_date'],
            ],
            asAt: Carbon::parse($item['as_at']),
            source: $item['source'],
            actor: $actor,
        );

    $update = $entry->learningUpdate()->firstOrFail();
    if ($update->status !== LearningUpdate::STATUS_IMPLEMENTED) {
        $update->forceFill([
            'status' => LearningUpdate::STATUS_APPROVED,
            'effective_date' => now()->subMinute(),
            'pre_implementation_notice_at' => now(),
            'review_due_at' => now()->addDays(30),
            'decided_by_user_id' => $actor->getKey(),
            'decided_at' => now(),
        ])->save();

        app(ApprovalFlow::class)->implementDue(now(), $actor);
    }
}
```
